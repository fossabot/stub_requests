def initialize(options = {})
  @service_id   = options[:service_id]
  @service_uri  = options[:service_uri]
  @id           = options[:endpoint_id]
  @verb         = options[:verb]
  @path         = options[:path]
  @uri          = URI.safe_join(service_uri, path)
  @route_params = URI.route_params(path)
end
