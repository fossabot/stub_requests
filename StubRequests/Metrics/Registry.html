def find_stub_stat(request_stub)
  map do |endpoint|
    endpoint.find_by(attribute: :request_stub, value: request_stub)
  end.compact.first
end

def record(service, endpoint, request_stub)
  stat = find_or_initialize_stat(service, endpoint)
  stat.record(request_stub)

  stats.push(stat)
  stat
end
