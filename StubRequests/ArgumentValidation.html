def validate_type!(name, value, type)
  expected_types = Array(type).flatten
  return if expected_types.any? { |is_a| value.is_a?(is_a) }

  raise StubRequests::InvalidArgumentType,
        name: name,
        actual: value.class,
        expected: expected_types
end
